pub type le16 = Le<u16>;
Expand description

A 16-bit unsigned integer stored in little-endian byte order.

§Examples

Basic usage:

use endian_num::le16;

let n = 0x1A;

if cfg!(target_endian = "little") {
    assert_eq!(le16::from_ne(n).0, n);
} else {
    assert_eq!(le16::from_ne(n).0, n.swap_bytes());
}
